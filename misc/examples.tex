% bib
@BOOK {key,
  title = {{name}}
  publisher = {name, ISBN}
  year = {YYYY}
  author = {name1 and name2 and name3}
}

@ARTICLE{key,
  title = {{name}},
  journal = {name},
  volume = {number},
  pages = {number--number},
  month = {MMM},
  year = {YYYY},
  author = {name1 and name2 ...},
}

% for anything else
@MISC{key,
  title       = {\emph{name}},
  author      = {{name}},
  lastchecked = {DD MMM YYYY},
  url         = {url}
}
